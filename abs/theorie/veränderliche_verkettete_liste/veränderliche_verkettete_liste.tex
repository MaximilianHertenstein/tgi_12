% -------------------------------------------------------
%  Hilfsmakros für verkettete Listen

\newif\ifllhasitems

% \drawlinkedlist{a,b,c}  – Standardliste mit first-Zeiger
% \drawlinkedlist{}        – leere Liste
\newcommand{\drawlinkedlist}[1]{%
  \global\llhasitemsfalse
  \gdef\llprev{first}%
  \begin{center}
  \begin{tikzpicture}[node distance=1.3cm]
    \node[firstptr] (first) {\texttt{first}};
    \foreach \llval [count=\lli] in {#1} {%
      \node[listnode, right=of \llprev] (llnode\lli)
          {\texttt{"\llval"} \nodepart{two} \phantom{xx}};
      \ifnum\lli=1\relax
        \draw[arrow] (first.east) -- (llnode1.west);
      \else
        \pgfmathtruncatemacro\llp{\lli-1}%
        \draw[arrow] (llnode\llp.two east) -- (llnode\lli.west);
      \fi
      \xdef\llprev{llnode\lli}%
      \global\llhasitemstrue
    }%
    \ifllhasitems
      \node[nullnode, right=of \llprev] (llnull) {\texttt{null}};
      \draw[arrow] (\llprev.two east) -- (llnull.west);
    \else
      \node[nullnode, right=2cm of first] (llnull) {\texttt{null}};
      \draw[arrow] (first.east) -- (llnull.west);
    \fi
  \end{tikzpicture}
  \end{center}%
}

% \drawlinkedlistix{a,b,c} – wie \drawlinkedlist, aber mit Indexbeschriftung
\newcommand{\drawlinkedlistix}[1]{%
  \global\llhasitemsfalse
  \gdef\llprev{first}%
  \begin{center}
  \begin{tikzpicture}[node distance=1.3cm]
    \node[firstptr] (first) {\texttt{first}};
    \foreach \llval [count=\lli] in {#1} {%
      \node[listnode, right=of \llprev] (llnode\lli)
          {\texttt{"\llval"} \nodepart{two} \phantom{xx}};
      \ifnum\lli=1\relax
        \draw[arrow] (first.east) -- (llnode1.west);
      \else
        \pgfmathtruncatemacro\llp{\lli-1}%
        \draw[arrow] (llnode\llp.two east) -- (llnode\lli.west);
      \fi
      \pgfmathtruncatemacro\llidx{\lli-1}%
      \node[below=0.3cm of llnode\lli, font=\small] {Index \llidx};
      \xdef\llprev{llnode\lli}%
      \global\llhasitemstrue
    }%
    \ifllhasitems
      \node[nullnode, right=of \llprev] (llnull) {\texttt{null}};
      \draw[arrow] (\llprev.two east) -- (llnull.west);
    \else
      \node[nullnode, right=2cm of first] (llnull) {\texttt{null}};
      \draw[arrow] (first.east) -- (llnull.west);
    \fi
  \end{tikzpicture}
  \end{center}%
}
